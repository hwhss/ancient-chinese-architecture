**aiService.js示例**：
```javascript
const axios = require('axios');

// 配置（替换为实际的API Key）
const API_KEY = 'your_api_key_here';
const API_URL = 'https://api.moonshot.cn/v1/chat/completions';  // Kimi API

// 系统提示词
const SYSTEM_PROMPT = `你是一个古建筑专业导览助手，仅回答指定古建筑核心景点的问题。
要求：
1. 回答简洁明了，不超过200字
2. 不回答无关问题
3. 如果用户问题不在知识库范围内，请礼貌告知`;

/**
 * 调用AI API获取回答
 */
async function getAIAnswer(question) {
  try {
    const response = await axios.post(
      API_URL,
      {
        model: 'moonshot-v1-8k',
        messages: [
          { role: 'system', content: SYSTEM_PROMPT },
          { role: 'user', content: question }
        ],
        temperature: 0.7
      },
      {
        headers: {
          'Authorization': `Bearer ${API_KEY}`,
          'Content-Type': 'application/json'
        }
      }
    );

    return response.data.choices[0].message.content;
  } catch (error) {
    console.error('AI API调用失败:', error);
    throw error;
  }
}

module.exports = { getAIAnswer };
```

**修改后的send方法**：
```javascript
import axios from 'axios';

// 本地关键词映射表（极简版）
const keywordMapping = {
  '太和殿': 'gugong_01',
  '乾清宫': 'gugong_02',
  '中和殿': 'gugong_03',
  '保和殿': 'gugong_04',
  '养心殿': 'gugong_05',
  '武英殿': 'gugong_06',
  '文华殿': 'gugong_07',
  '坤宁宫': 'gugong_08',
  '御花园': 'gugong_09',
  '神武门': 'gugong_10',
  '东华门': 'gugong_11',
  '西华门': 'gugong_12',
  '午门': 'gugong_13',
  '天安门': 'gugong_14',
  '端门': 'gugong_15'
};

// 根据问题匹配materialId
function matchMaterialId(question) {
  for (const keyword in keywordMapping) {
    if (question.includes(keyword)) {
      return keywordMapping[keyword];
    }
  }
  return null;  // 没有匹配到
}

async send() {
  if (!this.inputText.trim()) return;

  const question = this.inputText;
  this.inputText = '';

  // 添加用户消息
  this.messages.push({
    role: 'user',
    content: question
  });
  this.scrollToBottom();

  try {
    // 调用后端接口
    const response = await axios.post('http://localhost:3000/api/chat', {
      question: question
    });

    const answer = response.data.data.answer;
    const materialId = matchMaterialId(question);  // 本地匹配

    // 添加AI回复
    this.messages.push({
      role: 'ai',
      content: answer,
      materialId: materialId  // 可能为null
    });
    this.scrollToBottom();
  } catch (error) {
    console.error('接口调用失败:', error);
    this.messages.push({
      role: 'ai',
      content: '抱歉，请求失败，请重试'
    });
    this.scrollToBottom();
  }
}
```

app.post('/api/chat', async (req, res) => {

    if (!question) {
      return res.json({
        code: 400,
        msg: '问题不能为空'
      });
    }

    // 调用AI获取回答
    const answer = await getAIAnswer(question);

    // 返回回答（materialId由前端本地匹配）
    res.json({
      code: 200,
      msg: 'success',
      data: {
        answer: answer,
        question: question
      }
    });
  } catch (error) {
    console.error('问答接口错误:', error);
    res.json({
      code: 500,
      msg: '服务异常，请重试'
    });
  }
});
```

// 素材查询接口
app.get('/api/material', (req, res) => {
  try {
    const { materialId } = req.query;

    if (!materialId) {
      return res.json({
        code: 400,
        msg: 'materialId不能为空'
      });
    }

    // 查找素材
    const material = materialLinks.find(m => m.materialId === materialId);

    if (!material) {
      return res.json({
        code: 404,
        msg: '素材不存在'
      });
    }

    res.json({
      code: 200,
      msg: 'success',
      data: {
        url: material.url,
        type: material.type,
        source: material.source
      }
    });
  } catch (error) {
    console.error('素材查询接口错误:', error);
    res.json({
      code: 500,
      msg: '服务异常'
    });
  }
});
```
